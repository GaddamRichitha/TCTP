  1571→// ═══════════════════════════════════════════════════════════════════
  1572→// EARNED VALUE MANAGEMENT (EVM) ENGINE
  1573→// ═══════════════════════════════════════════════════════════════════
  1574→// PV (Planned Value)  = budgeted cost of work that SHOULD be done by current month
  1575→// EV (Earned Value)    = budgeted cost of work ACTUALLY done by current month (here: actual hours × planned rate)
  1576→// AC (Actual Cost)     = real cost incurred for hours actually worked (actual hours × rate)
  1577→// CV = EV − AC   (negative = over budget)
  1578→// SV = EV − PV   (negative = behind schedule)
  1579→// CPI = EV / AC  (>1 = under budget, <1 = over budget)
  1580→// SPI = EV / PV  (>1 = ahead of schedule, <1 = behind schedule)
  1581→// EAC = AC + (BAC − EV) / CPI   (forecast total cost at completion)
  1582→function computeEVM(d) {
  1583→  const duration     = d.duration;
  1584→  const currentMonth = Math.min(getNum('f-currentMonth', 1), duration);
  1585→  const stdHours      = getNum('f-stdHours', 160);
  1586→
  1587→  let PV = 0, EV = 0, AC = 0;
  1588→  let BAC = 0; // Budget at completion (total planned labour cost across full duration)
  1589→
  1590→  const perResource = [];
  1591→
  1592→  rows.labour.forEach(row => {
  1593→    const rate = parseFloat(row.rate) || 0;
  1594→    const qty  = parseFloat(row.qty)  || 1;
  1595→    const plannedHoursPerMonth = row.rateBasis === 'hourly'
  1596→      ? (parseFloat(row.plannedHours) || 0)
  1597→      : stdHours; // for monthly-salary resources, assume standard hours as the "planned" baseline
  1598→    const monthlyCost = getRowMonthlyCost(row); // full planned cost per month for this resource
  1599→    const impliedHourlyRate = plannedHoursPerMonth > 0 ? monthlyCost / plannedHoursPerMonth : 0;
  1600→
  1601→    let resPV = 0, resEV = 0, resAC = 0, planHoursToDate = 0, actualHoursToDate = 0;
  1602→
  1603→    for (let m = 1; m <= duration; m++) {
  1604→      const plannedHrsThisMonth = plannedHoursPerMonth;
  1605→      const isPast = m <= currentMonth;
  1606→
  1607→      // Planned value accrues every month up to current month regardless of actuals
  1608→      if (isPast) {
  1609→        resPV += monthlyCost * qty;
  1610→        planHoursToDate += plannedHrsThisMonth * qty;
  1611→      }
  1612→
  1613→      if (isPast) {
  1614→        const actualHrs = actualHours[row.id] && actualHours[row.id][m] !== undefined
  1615→          ? actualHours[row.id][m]
  1616→          : plannedHrsThisMonth; // unlogged past months assumed on-plan
  1617→        actualHoursToDate += actualHrs * qty;
  1618→
  1619→        // % of this month's planned work actually completed. Hours worked is our best
  1620→        // available proxy for progress: working fewer hours than planned means less of
  1621→        // the month's scope got done (schedule slip); working more hours doesn't earn
  1622→        // MORE value than was scoped — it just means it cost more to deliver the same
  1623→        // scope (cost overrun). Capped at 100% — you can't out-earn the month's scope.
  1624→        const pctComplete = plannedHrsThisMonth > 0 ? Math.min(actualHrs / plannedHrsThisMonth, 1.0) : 1.0;
  1625→        resEV += monthlyCost * qty * pctComplete;
  1626→
  1627→        // Actual cost = actual hours actually worked × implied hourly rate — this is
  1628→        // what really varies when someone works more or fewer hours than planned,
  1629→        // and is what diverges from EV to produce a genuine CPI signal.
  1630→        resAC += impliedHourlyRate * actualHrs * qty;
  1631→      }
  1632→    }
  1633→
  1634→    BAC += monthlyCost * qty * duration;
  1635→
  1636→    PV += resPV;
  1637→    EV += resEV;
  1638→    AC += resAC;
  1639→
  1640→    const variance = resEV - resAC;
  1641→    const status = variance >= 0 ? 'good' : (variance >= -resPV * 0.1 ? 'warn' : 'bad');
  1642→
  1643→    perResource.push({
  1644→      id: row.id,
  1645→      desc: row.desc || '(unnamed role)',
  1646→      planHoursToDate, actualHoursToDate,
  1647→      plannedCost: resPV, actualCost: resAC, earnedValue: resEV,
  1648→      variance, status
  1649→    });
  1650→
  1651→    // Update tracking page live cells
  1652→    const planEl   = document.getElementById('track-plan-'+row.id);
  1653→    const totalEl   = document.getElementById('track-total-'+row.id);
  1654→    const todateEl  = document.getElementById('track-todate-'+row.id);
  1655→    const statusEl  = document.getElementById('track-status-'+row.id);
  1656→    if (planEl)  planEl.textContent = plannedHoursPerMonth + ' hrs';
  1657→    if (totalEl) totalEl.textContent = Math.round(actualHoursToDate) + ' hrs';
  1658→    if (todateEl) todateEl.textContent = Math.round(planHoursToDate) + ' hrs';
  1659→    if (statusEl) {
  1660→      const diff = actualHoursToDate - planHoursToDate;
  1661→      const pct = planHoursToDate > 0 ? Math.round(diff/planHoursToDate*100) : 0;
  1662→      statusEl.innerHTML = diff === 0 ? '<span style="color:var(--ink3)">On plan</span>'
  1663→        : diff > 0 ? `<span style="color:var(--red)">+${pct}%</span>`
  1664→        : `<span style="color:var(--green)">${pct}%</span>`;
  1665→    }
  1666→  });
  1667→
  1668→  const CV  = EV - AC;
  1669→  const SV  = EV - PV;
  1670→  const CPI = AC > 0 ? EV / AC : 1;
  1671→  const SPI = PV > 0 ? EV / PV : 1;
  1672→  const EAC = CPI > 0 ? AC + (BAC - EV) / CPI : BAC;
  1673→  const ETC = EAC - AC; // estimate to complete
  1674→  const VAC = BAC - EAC; // variance at completion
  1675→
  1676→  // Non-labour costs (assumed to track on-plan since we don't track hours for them)
  1677→  const nonLabourTotal = CATS.filter(cat => cat !== 'labour').reduce((s, cat) => s + d.catTotals[cat].total, 0);
  1678→  const projectedTotalCost = EAC + nonLabourTotal;
  1679→  const originalTotalCost  = BAC + nonLabourTotal;
  1680→
  1681→  return {
  1682→    currentMonth, duration, stdHours,
  1683→    PV, EV, AC, BAC, EAC, ETC, VAC, CV, SV, CPI, SPI,
  1684→    perResource, nonLabourTotal, projectedTotalCost, originalTotalCost
  1685→  };
  1686→}

  1923→function computeCatTotals(duration) {
  1924→  const totals = {};
  1925→  CATS.forEach(cat => {
  1926→    totals[cat] = { monthly: 0, onetime: 0, perunit: 0, total: 0 };
  1927→    rows[cat].forEach(row => {
  1928→      const rate = parseFloat(row.rate) || 0;
  1929→      const qty  = parseFloat(row.qty)  || 1;
  1930→      if (row.type === 'monthly') {
  1931→        const monthlyCost = getRowMonthlyCost(row);
  1932→        totals[cat].monthly += monthlyCost * qty;
  1933→        totals[cat].total   += monthlyCost * qty * duration;
  1934→      } else if (row.type === 'onetime') {
  1935→        totals[cat].onetime += rate * qty;
  1936→        totals[cat].total   += rate * qty;
  1937→      } else if (row.type === 'perunit') {
  1938→        totals[cat].perunit += rate * qty;
  1939→      }
  1940→    });
  1941→  });
  1942→  return totals;
  1943→}
